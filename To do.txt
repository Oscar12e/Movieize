function binarySearch(pList, pValue) {
    var mid = Math.floor(pList.length / 2);
    console.log(pList[mid], pValue);

    if (pList[mid] === pValue) {
        console.log('match', pList[mid], pValue);
        return pList[mid];
    } else if (pList[mid] < pValue && pList.length > 1) {
        console.log('mid lower', pList[mid], pValue);
        binarySearch(pList.splice(mid, Number.MAX_VALUE), pValue);
    } else if (pList[mid] > pValue && pList.length > 1) {
        console.log('mid higher', pList[mid], pValue);
        binarySearch(pList.splice(0, mid), pValue);
    } else {
        console.log('not here', pValue);
        return -1;
    }

}
